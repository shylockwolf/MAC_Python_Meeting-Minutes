import mlx_whisper
import time

def transcribe_audio(audio_path, model="whisper-small-mlx", language="zh"):
    start_time = time.time()
    
    print(f"正在处理音频: {audio_path}")
    
    try:
        result = mlx_whisper.transcribe(
            audio_path,
            model=model,
            language=language,
            word_timestamps=True,
            fp16=True
        )
        
        processing_time = time.time() - start_time
        total_duration = result['segments'][-1]['end'] if result['segments'] else 0
        
        print(f"\n处理完成！")
        print(f"音频时长: {total_duration:.2f} 秒 ({total_duration/60:.2f} 分钟)")
        print(f"处理时间: {processing_time:.2f} 秒")
        print(f"处理速度: {total_duration/processing_time:.2f}x 实时速度")
        print(f"分段数量: {len(result['segments'])}")
        
        return result
        
    except Exception as e:
        print(f"处理失败: {e}")
        return None

# 使用示例
result = transcribe_audio("your_audio.mp3")

if result:
    print("\n--- 转录结果 ---")
    print(result["text"])
